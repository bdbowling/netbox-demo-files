#Show only the device list (results)
export NETBOX_TOKEN="PASTE_REAL_TOKEN_HERE"

curl -H "Authorization: Token $NETBOX_TOKEN" \
     https://yourinstance.cloud.netboxapp.com/api/dcim/devices/ \
  | python3 -m json.tool \
  | sed -n '/"results": \[/,/\]/p'
#(That keeps things readable even without jq.)

-----

#Device names only (clean, one per line) — jq
export NETBOX_TOKEN="PASTE_REAL_TOKEN_HERE"

curl -H "Authorization: Token $NETBOX_TOKEN" \
     https://yourinstance.cloud.netboxapp.com/api/dcim/devices/ \
  | jq -r '.results[].name'

-----

#Device name + role — jq
export NETBOX_TOKEN="PASTE_REAL_TOKEN_HERE"

curl -H "Authorization: Token $NETBOX_TOKEN" \
     https://yourinstance.cloud.netboxapp.com/api/dcim/devices/ \
  | jq '.results[] | {name: .name, role: .role.name}'

-----

#Device name + site + role (super common)
export NETBOX_TOKEN="PASTE_REAL_TOKEN_HERE"

curl -H "Authorization: Token $NETBOX_TOKEN" \
     https://yourinstance.cloud.netboxapp.com/api/dcim/devices/ \
  | jq '.results[] | {name: .name, site: .site.name, role: .role.name}'

-----

#Filter at the API level + pretty output (best practice)
#Example: only devices in site nyc

export NETBOX_TOKEN="PASTE_REAL_TOKEN_HERE"

curl -H "Authorization: Token $NETBOX_TOKEN" \
     "https://yourinstance.cloud.netboxapp.com/api/dcim/devices/?site=nyc" \
  | jq '.results[] | {name: .name, site: .site.name}'
